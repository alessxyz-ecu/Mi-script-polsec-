-- MEGA ULTRA LESS OPTIMIZER - Intro con texto GIGANTE EXTREMO (tamaño 680)

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")
local Lighting = game:GetService("Lighting")
local Workspace = game:GetService("Workspace")

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui", 5)

if not playerGui then return end

-- =============================================
-- Interfaz MEGA RÁPIDA + TEXTO ENTRADA GIGANTE EXTREMO
-- =============================================

task.defer(function()
    -- Intro ULTRA RÁPIDA con texto GIGANTESCO
    local introGui = Instance.new("ScreenGui")
    introGui.ResetOnSpawn = false
    introGui.Parent = playerGui

    local introText = Instance.new("TextLabel")
    introText.Size = UDim2.new(1.3, 0, 0.9, 0)          -- contenedor maximizado para texto masivo
    introText.AnchorPoint = Vector2.new(0.5, 0.5)
    introText.Position = UDim2.new(0.5, 0, 0.5, 0)
    introText.BackgroundTransparency = 1
    introText.Text = "LESS OPTIMIZER"
    introText.TextColor3 = Color3.fromRGB(255, 255, 255)
    introText.TextSize = 680                            -- ← ¡TAMAÑO 680! Gigante extremo
    introText.Font = Enum.Font.GothamBlack
    introText.TextXAlignment = Enum.TextXAlignment.Center
    introText.TextYAlignment = Enum.TextYAlignment.Center
    introText.TextStrokeTransparency = 0.15             -- stroke muy grueso para máxima visibilidad
    introText.TextStrokeColor3 = Color3.fromRGB(80, 180, 255)
    introText.TextTransparency = 1
    introText.Parent = introGui

    TweenService:Create(introText, TweenInfo.new(0.4), {TextTransparency = 0}):Play()

    local conn
    conn = RunService.RenderStepped:Connect(function()
        if introText.TextTransparency > 0.9 then conn:Disconnect() return end
        local t = tick() * 8
        introText.Position = UDim2.new(0.5, 0, 0.5, math.sin(t) * 100)  -- onda muy amplia para efecto épico
    end)

    task.delay(0.5, function()
        TweenService:Create(introText, TweenInfo.new(0.4), {TextTransparency = 1}):Play():Wait()
        introGui:Destroy()
    end)

    -- Barra superior (sin cambios)
    task.delay(0.8, function()
        local barGui = Instance.new("ScreenGui")
        barGui.ResetOnSpawn = false
        barGui.Parent = playerGui

        local barFrame = Instance.new("Frame")
        barFrame.Size = UDim2.new(0, 360, 0, 34)
        barFrame.Position = UDim2.new(0.5, -180, 0, -30)
        barFrame.BackgroundTransparency = 1
        barFrame.Parent = barGui

        local line = Instance.new("Frame")
        line.AnchorPoint = Vector2.new(0.5, 0.5)
        line.Position = UDim2.new(0.5, 0, 0, 0)
        line.Size = UDim2.new(1, 0, 0, 2)
        line.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        line.BorderSizePixel = 0
        line.Parent = barFrame

        local txt = Instance.new("TextLabel")
        txt.Size = UDim2.new(1, 0, 1, 0)
        txt.BackgroundTransparency = 1
        txt.Text = "LESS OPTIMIZER"
        txt.TextColor3 = Color3.fromRGB(255, 255, 255)
        txt.TextSize = 22
        txt.Font = Enum.Font.GothamBold
        txt.TextXAlignment = Enum.TextXAlignment.Center
        txt.TextYAlignment = Enum.TextYAlignment.Center
        txt.Parent = barFrame

        txt.TextTransparency = 1
        line.BackgroundTransparency = 1
        TweenService:Create(txt, TweenInfo.new(0.3), {TextTransparency = 0}):Play()
        TweenService:Create(line, TweenInfo.new(0.3), {BackgroundTransparency = 0}):Play()

        local amp = 0.055
        local spd = 2.5
        RunService.RenderStepped:Connect(function()
            local s = 1 + math.sin(tick() * spd) * amp
            line.Size = UDim2.new(s, 0, 0, 2)
        end)
    end)
end)

-- =============================================
-- OPTIMIZACIÓN MEGA RÁPIDA (sin cambios)
-- =============================================

local function megaOptimize()
    pcall(function()
        Lighting.GlobalShadows = false
        Lighting.Brightness = 1.2
        Lighting.Ambient = Color3.fromRGB(130, 130, 130)
        Lighting.OutdoorAmbient = Color3.fromRGB(110, 110, 110)
        Lighting.ShadowSoftness = 0
        Lighting.EnvironmentDiffuseScale = 0
        Lighting.EnvironmentSpecularScale = 0
        Lighting.ClockTime = 12

        for _, v in Lighting:GetChildren() do
            if v:IsA("PostEffect") then v.Enabled = false end
        end

        local function optimizeObj(obj)
            if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Sparkles") or obj:IsA("Smoke") or obj:IsA("Fire") then
                obj.Enabled = false
            elseif obj:IsA("Light") or obj:IsA("PointLight") or obj:IsA("SpotLight") or obj:IsA("SurfaceLight") then
                obj.Enabled = false
            elseif obj:IsA("BasePart") or obj:IsA("MeshPart") or obj:IsA("UnionOperation") then
                obj.CastShadow = false
                if obj.Transparency < 0.05 then
                    obj.Material = Enum.Material.SmoothPlastic
                    obj.Reflectance = 0
                end
            end
        end

        task.defer(function()
            for _, obj in Workspace:GetDescendants() do
                optimizeObj(obj)
            end
        end)

        Workspace.DescendantAdded:Connect(optimizeObj)

        Workspace.StreamingEnabled = true
        Workspace.StreamingMinRadius = 32
        Workspace.StreamingTargetRadius = 96

        settings().Rendering.QualityLevel = Enum.QualityLevel.Level01

        print("[MEGA OPTIMIZER] Activado - ¡Super rápido!")
    end)
end

megaOptimize()

player.CharacterAdded:Connect(megaOptimize)
